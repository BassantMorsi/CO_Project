module dm(
		input  wire		clk,
		input  wire	[6:0]	addr,
		input  wire	        rd, wr,
		input  wire 	[31:0]	wdata,
		output wire	[31:0]	rdata);
 
	reg [31:0] mem [0:127];  

always @(posedge clk) begin
		if (wr) begin
			mem[addr] <= wdata;
		end
	end
  

	assign rdata = wr ? wdata : mem[addr][31:0];
      
	

endmodule
